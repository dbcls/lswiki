#!/usr/bin/env ruby

require 'rubygems'
require 'json'
require 'json/ld'

jsonld = "./NC_000911.gb.gff.ldj"
graph = RDF::Graph.new
i = 0
File.open(jsonld).each do |line|
  input = JSON.parse line
  #JSON::LD::API.toRDF(input){|st| p st} 
  graph << JSON::LD::API.toRdf(input)
end

require 'rdf/turtle'
